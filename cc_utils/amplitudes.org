BEGIN_PROVIDER [double precision, t1, (dim_list_act_orb, dim_list_virt_orb)]

  implicit none

  integer :: i,a
  integer :: nO, nV
  
  BEGIN_DOC
  ! t1_{i}^{a} amplitudes
  ! i: occupied MOs
  ! a: virtual MOs
  END_DOC

  nO = dim_list_act_orb
  nV = dim_list_virt_orb

  do a = 1, nV
    do i = 1, nO
      t1(i,a) = 0d0
    enddo
  enddo

END_PROVIDER
