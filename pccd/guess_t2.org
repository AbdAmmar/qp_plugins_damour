Guess for $t$
$$t_i^a = \frac{v_{ii}^{aa}}{(2f_a^a - 2f_i^i)}$$


#+BEGIN_SRC f90 :comments org :tangle guess_t2.irp.f
subroutine guess_amplitudes_pccd(t2)
  
  implicit none

  ! out
  double precision, intent(out) :: t2(dim_list_act_orb, dim_list_virt_orb)

  ! internal
  integer                       :: i, a
  integer                       :: nO, nV

  nO = dim_list_act_orb
  nV = dim_list_virt_orb

  ! Guess for t2 = t_i^a / (2f_a^a - 2f_i^i)
  do a = 1, nV
    do i = 1, nO
      t2(i,a) = - two_e_ints_iiaa(i,a) &
        /(2d0 * f_v(a) - 2d0 * f_o(i))
      !print*,i,a,t2(i,a)
    enddo
  enddo

end  
#+END_SRC
